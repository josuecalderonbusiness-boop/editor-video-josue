<script>
  // Timings basados en el SRT:
  // s1: 2:29 (inicio) → dura 3s
  // s2: 2:32 → dura 5s
  // s3: 2:37 → dura 8s
  // s4: 2:45 → dura 9s
  // s5: 2:54 → dura 11s hasta ~3:05

  const SLIDES = [
    { id:'s1', dur:3000  },
    { id:'s2', dur:5000  },
    { id:'s3', dur:8000  },
    { id:'s4', dur:9000  },
    { id:'s5', dur:11000 },
  ];
  const TRANS = 350;
  let current = 0;

  function show(idx) {
    const el = document.getElementById(SLIDES[idx].id);
    el.style.opacity = '1';
    el.style.pointerEvents = 'auto';
  }

  function hide(idx) {
    const el = document.getElementById(SLIDES[idx].id);
    el.style.animation = `cardOut ${TRANS}ms ease-in forwards`;
    setTimeout(() => {
      el.style.opacity = '0';
      el.style.pointerEvents = 'none';
      el.style.animation = '';
    }, TRANS);
  }

  function next() {
    hide(current);
    current++;
    if (current < SLIDES.length) {
      setTimeout(() => {
        show(current);
        if (current < SLIDES.length - 1) {
          setTimeout(next, SLIDES[current].dur);
        }
      }, TRANS);
    }
  }

  show(0);
  setTimeout(next, SLIDES[0].dur);
</script>
