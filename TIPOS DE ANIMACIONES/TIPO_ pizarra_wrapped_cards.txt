<script>
  const SLIDES = [
    { id:'s0',     dur:2500 },
    { id:'s1',     dur:4200 },
    { id:'s2',     dur:4200 },
    { id:'s3',     dur:4200 },
    { id:'s4',     dur:4200 },
    { id:'sfinal', dur:6000 },
  ];
  const TRANS = 380;
  let current = 0;

  function show(idx) {
    const el = document.getElementById(SLIDES[idx].id);
    el.style.opacity = '1';
    el.style.pointerEvents = 'auto';
  }

  function hide(idx) {
    const el = document.getElementById(SLIDES[idx].id);
    el.style.animation = `cardOut ${TRANS}ms ease-in forwards`;
    setTimeout(() => {
      el.style.opacity = '0';
      el.style.pointerEvents = 'none';
      el.style.animation = '';
    }, TRANS);
  }

  function next() {
    hide(current);
    current++;
    if (current < SLIDES.length) {
      setTimeout(() => {
        show(current);
        if (current < SLIDES.length - 1) {
          setTimeout(next, SLIDES[current].dur);
        }
      }, TRANS);
    }
  }

  show(0);
  setTimeout(next, SLIDES[0].dur);
</script>
